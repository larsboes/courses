Quick Reference

Bash kubectl commands 
# Deployen
kubectl apply -k overlays/uebung-12/

# Status
kubectl -n fhdw-wta get all

# Logs
kubectl -n fhdw-wta logs -l component=webserver

# Loeschen
kubectl delete -k overlays/uebung-12/

Bash debug commands 
# Pod Details
kubectl -n fhdw-wta describe pod <pod-name>

# In Container einloggen
kubectl -n fhdw-wta exec -it <pod-name> -- /bin/sh

# Events anzeigen
kubectl -n fhdw-wta get events --sort-by=.lastTimestamp

# Service Endpoints
kubectl -n fhdw-wta get endpoints
